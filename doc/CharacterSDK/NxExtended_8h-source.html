00644             {
00645             min = c;    min -= e;
00646             max = c;    max += e;
00647             }

00649         NX_INLINE void  getCenter(NxExtendedVec3& center) const
00650             {
00651             center = min + max;
00652             center *= 0.5;

00656             {

00671         NX_INLINE bool  intersect(const NxExtendedBounds3& b) const
00672             {
00673             if ((b.min.x > max.x) || (min.x > b.max.x)) return false;
00674             if ((b.min.y > max.y) || (min.y > b.max.y)) return false;
00675             if ((b.min.z > max.z) || (min.z > b.max.z)) return false;
00676             return true;

00679         NX_INLINE void  boundsOfOBB(const NxMat33& orientation, const NxExtendedVec3& translation, const NxVec3& halfDims)
00680             {
00681             NxReal dimx = halfDims[0];
00682             NxReal dimy = halfDims[1];
00683             NxReal dimz = halfDims[2];
00684 
00685             NxReal x = NxMath::abs(orientation(0,0) * dimx) + NxMath::abs(orientation(0,1) * dimy) + NxMath::abs(orientation(0,2) * dimz);
00686             NxReal y = NxMath::abs(orientation(1,0) * dimx) + NxMath::abs(orientation(1,1) * dimy) + NxMath::abs(orientation(1,2) * dimz);
00687             NxReal z = NxMath::abs(orientation(2,0) * dimx) + NxMath::abs(orientation(2,1) * dimy) + NxMath::abs(orientation(2,2) * dimz);
00688 
00689             set(Extended(-x) + translation[0], Extended(-y) + translation[1], Extended(-z) + translation[2], Extended(x) + translation[0], Extended(y) + translation[1], Extended(z) + translation[2]);
00690             }

00692         NX_INLINE void  transform(const NxMat33& orientation, const NxExtendedVec3& translation)
00693             {
00694             // convert to center and extents form
00695             NxExtendedVec3 center;
00696             NxVec3 extents;
00697             getCenter(center);
00698             getExtents(extents);
00699 
00700 //          center = orientation * center + translation;
00701             NxMat33_multiply(orientation, center, center);
00702             center += translation;
00703 
00704             boundsOfOBB(orientation, center, extents);
00705             }

00707         NX_INLINE void  add(const NxExtendedBounds3& b2)
00708             {
00709             // - if we're empty, min = MAX,MAX,MAX => min will be b2 in all cases => it will copy b2, ok
00710             // - if b2 is empty, the opposite happens => keep us unchanged => ok
00711             // => same behaviour as before, automatically
00712             min.min(b2.min);
00713             max.max(b2.max);
00714             }

00716         NX_INLINE bool  isInside(const NxExtendedBounds3& box) const
00717             {
00718             if(box.min.x > min.x)   return false;
00719             if(box.min.y > min.y)   return false;
00720             if(box.min.z > min.z)   return false;
00721             if(box.max.x < max.x)   return false;
00722             if(box.max.y < max.y)   return false;
00723             if(box.max.z < max.z)   return false;
00724             return true;
00725             }

00730             NxVec3 extents; getExtents(extents);
00731             setCenterExtents(center, extents * scale);
