00715 NX_INLINE bool NxFluidDescBase::isValid() const
00716     {
00717     if (kernelRadiusMultiplier < 1.0f) return false;
00718     if (restDensity <= 0.0f) return false;
00719     if (restParticlesPerMeter <= 0.0f) return false;
00720 
00721     if (packetSizeMultiplier < 4) return false;
00722     if (packetSizeMultiplier & ( packetSizeMultiplier - 1 ) ) return false; 
00723 
00724     if (motionLimitMultiplier <= 0.0f) return false;
00725     if (motionLimitMultiplier > packetSizeMultiplier*kernelRadiusMultiplier) return false;
00726 
00727     if (collisionDistanceMultiplier <= 0.0f) return false;
00728     if (collisionDistanceMultiplier > packetSizeMultiplier*kernelRadiusMultiplier) return false;
00729 
00730     if (stiffness <= 0.0f) return false;
00731     if (viscosity <= 0.0f) return false;
00732     if (surfaceTension < 0.0f) return false;
00733 
00734     bool isNoInteraction = (simulationMethod & NX_F_NO_PARTICLE_INTERACTION) > 0;
00735     bool isSPH = (simulationMethod & NX_F_SPH) > 0;
00736     bool isMixed = (simulationMethod & NX_F_MIXED_MODE) > 0;
00737     if (!(isNoInteraction || isSPH || isMixed)) return false;
00738     if (isNoInteraction && (isSPH || isMixed)) return false;
00739     if (isSPH && (isNoInteraction || isMixed)) return false;
00740     if (isMixed && (isNoInteraction || isSPH)) return false;
00741     
00742     if (damping < 0.0f) return false;
00743     if (fadeInTime < 0.0f) return false;
00744 
00745     if (projectionPlane.normal.isZero()) return false;
00746 
00747     if (dynamicFrictionForDynamicShapes < 0.0f || dynamicFrictionForDynamicShapes > 1.0f) return false;
00748     if (staticFrictionForDynamicShapes < 0.0f || staticFrictionForDynamicShapes > 1.0f) return false;
00749     if (restitutionForDynamicShapes < 0.0f || restitutionForDynamicShapes > 1.0f) return false;
00750     if (attractionForDynamicShapes < 0.0f) return false;
00751     if (dynamicFrictionForStaticShapes < 0.0f || dynamicFrictionForStaticShapes > 1.0f) return false;
00752     if (staticFrictionForStaticShapes < 0.0f || staticFrictionForStaticShapes > 1.0f) return false;
00753     if (restitutionForStaticShapes < 0.0f || restitutionForStaticShapes > 1.0f) return false;
00754     if (attractionForStaticShapes < 0.0f) return false;
00755     if (collisionResponseCoefficient < 0.0f) return false;
00756     
00757     if (!initialParticleData.isValid()) return false;
00758     if (!particlesWriteData.isValid()) return false;
00759     if (!particleDeletionIdWriteData.isValid()) return false;
00760     if (!particleCreationIdWriteData.isValid()) return false;
