00400 bool NxD6JointDesc::isValid() const
00401 {
00402     if (flags & NX_D6JOINT_SLERP_DRIVE) {   // only possible with all angular DOF available
00403         if (swing1Motion == NX_D6JOINT_MOTION_LOCKED ||
00404             swing2Motion == NX_D6JOINT_MOTION_LOCKED ||
00405             twistMotion == NX_D6JOINT_MOTION_LOCKED) return false;
00406     }
00407 
00408     // swing limits are symmetric, thus their range is 0..180 degrees
00409     if (swing1Motion == NX_D6JOINT_MOTION_LIMITED) {
00410         if (swing1Limit.value < 0.0f) return false;
00411         if (swing1Limit.value > NxPi) return false;
00412     }
00413     if (swing2Motion == NX_D6JOINT_MOTION_LIMITED) {
00414         if (swing2Limit.value < 0.0f) return false;
00415         if (swing2Limit.value > NxPi) return false;
00416     }
00417 
00418     // twist limits are asymmetric with -180 <= low < high <= 180 degrees
00419     if (twistMotion == NX_D6JOINT_MOTION_LIMITED) {
00420         if (twistLimit.low.value < -NxPi) return false;
00421         if (twistLimit.high.value > NxPi) return false;
00422         if (twistLimit.low.value > twistLimit.high.value) return false;
00423     }
00424 
00425     // in angular limited-free mode, only -90..90 swings are possible
00426     if (swing1Motion == NX_D6JOINT_MOTION_LIMITED && swing2Motion == NX_D6JOINT_MOTION_FREE) 
00427         if (swing1Limit.value > NxHalfPi) return false; 
00428     if (swing2Motion == NX_D6JOINT_MOTION_LIMITED && swing1Motion == NX_D6JOINT_MOTION_FREE) 
00429         if (swing2Limit.value > NxHalfPi) return false; 
00430 
00431     if (flags & NX_D6JOINT_GEAR_ENABLED)        // gear only with twist motion enabled
00432         if (twistMotion == NX_D6JOINT_MOTION_LOCKED) return false;
00433 
00434     return NxJointDesc::isValid();
00435 }
