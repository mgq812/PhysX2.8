01710             0           5                   2           4                   1           3

01712 
01713     NxReal d = b00*data.s._11       +       b01*data.s._21               + b02 * data.s._31;
01714

01716         {
01717         dest.id();

01719         }
01720     
01721     d = NxReal(1.0)/d;
01722

01727     dest.data.s._21 = b10*d; dest.data.s._22 = b11*d; dest.data.s._23 = b12*d;
01728     dest.data.s._31 = b20*d; dest.data.s._32 = b21*d; dest.data.s._33 = b22*d;
01729 
01730     return true;
01731     }
