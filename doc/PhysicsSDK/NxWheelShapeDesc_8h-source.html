00394 NX_INLINE NxReal NxTireFunctionDesc::hermiteEval(NxReal t) const
00395     {
00396     
00397     // This fix for TTP 3429 & 3675 is from Sega.
00398     // Assume blending functions (look these up in a graph):
00399     // H0(t) =  2ttt - 3tt + 1
00400     // H1(t) = -2ttt + 3tt
00401     // H2(t) =   ttt - 2tt + t
00402     // H3(t) =   ttt -  tt 
00403 
00404     NxReal v = NxMath::abs(t);
00405     NxReal s = (t>=0) ? 1.0f : -1.0f;
00406 
00407     NxReal F;
00408 
00409     if(v<extremumSlip)
00410         {
00411         // For t in the interval 0 < t < extremumSlip
00412         // We normalize t:
00413         // a = t/extremumSlip;
00414         // and use H1 + H2 to compute F:
00415         // F = extremumValue * ( H1(a) + H2(a) )
00416 
00417         NxReal a = v/extremumSlip;
00418         NxReal a2 = a*a;
00419         NxReal a3 = a*a2;
00420 
00421         F = extremumValue * (-a3 + a2 + a);
00422         }
00423     else
00424         {
00425         if(v<asymptoteSlip)
00426             {
00427             // For the interval extremumSlip <= t < asymtoteSlip
00428             // We normalize and remap t:
00429             // a = (t-extremumSlip)/(asymptoteSlip - extremumSlip)
00430             // and use H0 to compute F:
00431             // F = extremumValue + (extremumValue - asymtoteValue) * H0(a)
00432             // note that the above differs from the actual expression but this is how it looks with H0 factorized.
00433                 
00434             NxReal a = (v-extremumSlip)/(asymptoteSlip - extremumSlip);
00435             NxReal a2 = a*a;
00436             NxReal a3 = a*a2;
00437 
00438             NxReal diff = asymptoteValue - extremumValue;
00439             F = -2.0f*diff*a3 + 3.0f*diff*a2 + extremumValue;
00440             }
00441         else
00442             {
00443             F = asymptoteValue;
00444             }
00445         }
00446     return s*F;
00447     }
