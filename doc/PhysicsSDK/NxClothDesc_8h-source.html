00789

00792     if(thickness < 0.0f) return false;

00797     if(tearFactor <= 1.0f) return false;

00800     if(friction < 0.0f || friction > 1.0f) return false;
00801     if(!meshData.isValid()) return false;
00802     if(dampingCoefficient < 0.0f || dampingCoefficient > 1.0f) return false;
00803     if(collisionResponseCoefficient < 0.0f) return false;
